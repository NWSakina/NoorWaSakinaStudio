// أضف هذا الجزء في دالة التحريك لضمان سلاسة الإطار const container = document.getElementById('text-container'); container.style.touchAction = 'none'; // لضمان عدم تعارض اللمس container.addEventListener('pointermove', (e) => { if (e.pointerType === 'touch') { e.preventDefault(); container.style.left = (e.clientX - container.offsetWidth/2) + 'px'; container.style.top = (e.clientY - container.offsetHeight/2) + 'px'; } }, {passive: false});
